Implement a Bulletproof vite.config.js AlternativeIf the custom regex string replace function inside the moveScriptToBody plugin is throwing errors, replace its injection hook with a safer, native AST modification method or a cleaner inline string replacement:javascript// A safer pattern for moveScriptToBody inside vite.config.js
export default {
  // ... rest of your config
  plugins: [
    {
      name: 'move-script-to-body-safe',
      transformIndexHtml(html) {
        // Find the injected module script tag cleanly
        const scriptRegex = /<script type="module" crossorigin src="\/assets\/index-.*?\.js"><\/script>/;
        const match = html.match(scriptRegex);
        
        if (!match) return html;
        
        // Strip it from its current position
        let cleanHtml = html.replace(scriptRegex, '');
        
        // Inject it safely right before the closing body tag
        return cleanHtml.replace('</body>', `${match[0]}\n</body>`);
      }
    }
  ]
}
