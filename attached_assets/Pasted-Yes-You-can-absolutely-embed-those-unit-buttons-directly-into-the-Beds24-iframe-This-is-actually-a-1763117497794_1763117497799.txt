3. Head Bottom (Beds24: "Head Bottom" section)
<script>
(function() {
  // Wait for DOM to be ready
  function initUnitButtons() {
    const buttons = document.querySelectorAll('.unit-btn-beds24');
    if (buttons.length === 0) return;
    buttons.forEach(function(btn) {
      btn.addEventListener('click', function() {
        const roomId = this.getAttribute('data-roomid');
        
        // Remove active class from all buttons
        buttons.forEach(function(b) {
          b.classList.remove('active');
        });
        
        // Add active class to clicked button
        this.classList.add('active');
        
        // Update the booking form to show selected room
        // This triggers Beds24's room filter
        const roomSelect = document.querySelector('select[name="roomid"]');
        if (roomSelect) {
          roomSelect.value = roomId;
          // Trigger change event to update availability
          const event = new Event('change', { bubbles: true });
          roomSelect.dispatchEvent(event);
        }
        
        // Also update URL parameter for Beds24's internal filtering
        const url = new URL(window.location.href);
        url.searchParams.set('roomid', roomId);
        window.history.replaceState({}, '', url.toString());
        
        // Reload booking form with selected room
        if (typeof b24_reload_page !== 'undefined') {
          b24_reload_page();
        } else {
          window.location.href = url.toString();
        }
      });
    });
    
    // Check for preselected room in URL
    const urlParams = new URLSearchParams(window.location.search);
    const preselectedRoom = urlParams.get('roomid');
    if (preselectedRoom) {
      const activeBtn = document.querySelector('.unit-btn-beds24[data-roomid="' + preselectedRoom + '"]');
      if (activeBtn) {
        activeBtn.classList.add('active');
      }
    }
  }
  
  // Initialize when DOM is ready
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', initUnitButtons);
  } else {
    initUnitButtons();
  }
})();
</script>
